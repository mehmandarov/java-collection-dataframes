public String outputToJson(Function0<Object> function) {
    Object result = function.value();
    try {
        return this.getObjectMapper().writeValueAsString(
                switch (result) {
                    case Bag<?> b -> b.toMapOfItemToCount();
                    case Multimap<?, ?> m -> m.toMap(Sets.mutable::empty);
                    default -> result;
                });
    }
